1    import traceback
2    import sys
3    from discord.ext import commands
4    import discord

7    class CommandErrorHandler(commands.Cog):
8        def __init__(self, bot):
9            self.bot = bot
10   
11       @commands.Cog.listener()
12       async def on_command_error(self, ctx, error):
13           """The event triggered when an error is raised while invoking a command. 
14           ctx   : Context 
15           error : Exception"""
16   
17           if hasattr(ctx.command, 'on_error'):
18               return
19   
20           ignored = (commands.UserInputError)
21           error = getattr(error, 'original', error)
22   
23           if isinstance(error, ignored):
24               return
25   
26           elif isinstance(error, commands.DisabledCommand):
27               return await ctx.send(f'{ctx.command} has been disabled.')
28   
29           elif isinstance(error, commands.NoPrivateMessage):
30               try:
31                   return await ctx.author.send(f'{ctx.command} can not be used in Private Messages.')
32               except:
33                   pass
34   
35           elif isinstance(error, commands.BadArgument):
36               if ctx.command.qualified_name == 'tag list':
37                   return await ctx.send('I could not find that member. Please try again.')
38   
39           elif isinstance(error, commands.CommandNotFound):
40               await ctx.send("Sorry the command does not exist.\nSay \'help\' to get the commands")
41               
42   
43           print('Ignoring exception in command {}:'.format(ctx.command), file=sys.stderr)
44           traceback.print_exception(type(error), error, error.__traceback__, file=sys.stderr)
45   
46       @commands.command(name='repeat', aliases=['mimic', 'copy'],hidden=True)
47       @commands.is_owner()
48       async def do_repeat(self, ctx, *, inp: str):
49           await ctx.send(inp)
50   
51       @do_repeat.error
52       async def do_repeat_handler(self, ctx, error):
53           if isinstance(error, commands.MissingRequiredArgument) and error.param.name == 'inp' :
54               await ctx.send("You forgot to give me input to repeat!")
55   
56   
57   def setup(bot):
58       bot.add_cog(CommandErrorHandler(bot))
