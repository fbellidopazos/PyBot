BOT_PREFIX = (".", "")
client = commands.Bot(command_prefix=BOT_PREFIX, description="Discord Python Rewrite Bot")

Users = []


def find(nick):
    res = -1
    for i in range(len(Users)):
        res += 1
        if (Users[i].nick == nick):
            break
    return res


def backUpPickle():
    f = open("Users.pkl", "wb")
    pickle.dump(Users, f)
    f.close()


def loadPickle():
    f = open("Users.pkl", "rb")
    Users = pickle.load(f)
    f.close()


def usersToString():
    res = ""
    for i in Users:
        res = res + i.toString()
    return (res)
